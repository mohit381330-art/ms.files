#include<iostream>
using namespace std;
int main() {
    int a=55;
    int*ptr=&a;
    cout<<ptr<<endl;        //value of ptr
    cout<<&ptr<<endl;       //address of ptr
    cout<<*ptr<<endl;       //value at ptr
    cout<<&a<<endl;         // address of a
    return 0;
}

//pointer to pointer
#include<iostream>
using namespace std;
int main() {
    int a=10;
    int *ptr1=&a;
    int **ptr2=&ptr1;
    cout<<ptr1<<endl;
    cout<<&ptr1<<endl;
    cout<<*ptr1<<endl;
    cout<<ptr2<<endl;
    cout<<&ptr2<<endl;
    cout<<*ptr2<<endl;
    cout<<**ptr2<<endl;
    return 0;
}
