//pointer
#include<iostream>
using namespace std;
int main() {
    int a=55;
    int*ptr=&a;
    cout<<a<<endl;
    cout<<&a<<endl;
    cout<<ptr<<endl;        //value of ptr
    cout<<&ptr<<endl;       //address of ptr
    cout<<*ptr<<endl;       //value at ptr        
    return 0;
}

//pointer to pointer
#include<iostream>
using namespace std;
int main() {
    int a=55;
    int *ptr1=&a;
    int **ptr2=&ptr1;
    cout<<&a<<endl;
    cout<<ptr1<<endl;
    cout<<&ptr1<<endl;
    cout<<*ptr1<<endl;
    cout<<ptr2<<endl;
    cout<<&ptr2<<endl;
    cout<<*ptr2<<endl;
    cout<<**ptr2<<endl;
    // dereference operator(*)
    cout<<*(&a)<<endl;
    cout<<*(ptr1)<<endl;
    cout<<*(ptr2)<<endl;
    cout<<**(ptr2)<<endl;

    //Null pointer
    #include<iostream>
    using namespace std;
    int main() {
    int *ptr=NULL;
    cout<<ptr<<endl;
    return 0;
}

//Pass by reference
#include<iostream>
using namespace std;
void changeA(int*ptr) {
    *ptr=20;
}
int main() {
    int a=10;
    changeA(&a);
    cout<<"inside main fxn:"<<a<<endl;
    return 0;
}

//pass by alias
#include<iostream>
using namespace std;
void changeB(int &b) {
    b=20;
}
int main() {
    int a=10;
    changeB(a);
    cout<<"inside main fxn:"<<a<<endl;
    return 0;
}

//array pointer and pointer arithmetic
//pointer arithmetic
#include<iostream>
using namespace std;

    int arr[5]={1,2,3,4,5,};

    cout<<ptr<<endl;

    cout<<*(ptr+1)<<endl;

    cout<<ptr<<endl;
    ptr++;
