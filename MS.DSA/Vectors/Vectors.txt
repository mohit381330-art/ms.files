// Vector Syntax
#include <iostream>
#include <vector>
using namespace std;
int main() {
    vector<int> vec = {1, 2, 3};
    cout << vec[2] << endl;
    return 0;
}

// Loops in Vector
#include <iostream>
#include <vector>
using namespace std;
int main() {
    vector<int> vec = {1, 2, 3, 4, 5, 6};
    for (int i : vec) {
        cout << i << endl;
    }
    return 0;
}

// Single Number
#include <iostream>
using namespace std;
int main() {
    int ans = 0;
    int arr[] = {2, 2, 2, 4};
    for (int i : arr) {
        ans ^= i;
    }
    cout << ans << endl;
    return 0;
}
