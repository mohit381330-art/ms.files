// Vector Syntax
#include<iostream>
#include<vector>
using namespace std;
int main() {
   vector<int> vec={1,2,3};
   cout<<vec[2]<<endl;
   return 0;
}

// loops in vector
#include<iostream>
#include<vector>
using namespace std;
int main() {
    vector<int> vec{1,2,3,4,5,6};
    for(int i:vec) {
        cout<<i<<endl;
    }
    return 0;
}

// Single Number
#include<iostream>
